    /**
     * Creates new form MainFrame

    public MainFrame() {
        initComponents();
        levelSelect.removeAllItems();
        for(int i = 1; i<=maxLevels;i++)
        {
            levelSelect.addItem("Level " + i);
        }
        
        board = new Board(level);
        player = board.player;
        this.repaintTimer = new Timer(300, repainter);
        repaintTimer.start();
        drawPanel.setBoard(board);
        
        
        if(!debugMode)
        {
            turnButton.setVisible(false);
            moveButton.setVisible(false);
            attackButton.setVisible(false);
            collectButton.setVisible(false);
        }
        //this.setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE);
    }
    ActionListener repainter = new ActionListener() {
        public void actionPerformed(ActionEvent evt) {
          drawPanel.repaint();
          if(board.isOver())
          {
            repaintTimer.stop();
          }
        }
    };
    
    
    public void reset()
    {
        level = levelSelect.getSelectedIndex()+1;
        board.reset(level);
        repaintTimer.start();
    }
    public Object getInterfaceObject()
    {
        return player;
    }
